## Task 4: Typography Hierarchy Verification

**Action**: Create `test/theme/typography_hierarchy_test.dart`.

```dart
group('Typography Hierarchy — 3+ distinct levels', () {
  test('semantic styles have strictly decreasing font sizes', () {
    final sizes = [
      AppTypography.sectionTitle.fontSize!,    // 24
      AppTypography.subsectionTitle.fontSize!,  // 18
      AppTypography.fieldValue.fontSize!,       // 16
      AppTypography.fieldLabel.fontSize!,       // 14
      AppTypography.statusBadge.fontSize!,      // 12
    ];
    for (int i = 0; i < sizes.length - 1; i++) {
      expect(sizes[i], greaterThan(sizes[i + 1]),
        reason: 'Size at index $i (${sizes[i]}) should be > index ${i + 1} (${sizes[i + 1]})');
    }
  });

  test('at least 4 distinct font sizes in semantic styles', () {
    final sizes = {
      AppTypography.sectionTitle.fontSize,
      AppTypography.subsectionTitle.fontSize,
      AppTypography.fieldValue.fontSize,
      AppTypography.fieldLabel.fontSize,
      AppTypography.statusBadge.fontSize,
      AppTypography.timestamp.fontSize,
    };
    expect(sizes.length, greaterThanOrEqualTo(4));
  });

  test('all semantic styles have minimum 11sp for outdoor legibility', () {
    final styles = [
      AppTypography.sectionTitle,
      AppTypography.subsectionTitle,
      AppTypography.fieldValue,
      AppTypography.fieldLabel,
      AppTypography.fieldLabelRequired,
      AppTypography.sectionHeader,
      AppTypography.statusBadge,
      AppTypography.timestamp,
    ];
    for (final style in styles) {
      expect(style.fontSize ?? 0, greaterThanOrEqualTo(11.0),
        reason: 'Style with size ${style.fontSize} is below 11sp minimum');
    }
  });

  test('sectionTitle has highest weight (w700)', () {
    expect(AppTypography.sectionTitle.fontWeight, equals(FontWeight.w700));
  });

  test('fieldLabelRequired uses primary color', () {
    expect(AppTypography.fieldLabelRequired.color, equals(Palette.primary));
  });
});
```

**Verify**: `flutter test test/theme/typography_hierarchy_test.dart` — all pass.
